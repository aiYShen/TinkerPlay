#!/bin/bash

# 脚本名称: call_multiple_springboot_apis.sh
# 功能: 定时调用多个Spring Boot接口并记录日志

# 日志文件路径
LOG_FILE="/var/log/springboot_multi_api_call.log"
# 请求超时时间（秒）
TIMEOUT=10
# 失败重试次数
RETRY_COUNT=3
# 重试间隔（秒）
RETRY_DELAY=5

# 接口配置数组
# 格式: "HTTP_METHOD|URL|BODY（仅POST需要，GET留空）"
API_CALLS=(
    "GET|http://localhost:8080/api/endpoint1|"
    "POST|http://localhost:8081/api/endpoint2|{\"key\":\"value\"}"
    "GET|http://localhost:8082/api/endpoint3|"
)

# 确保日志目录存在
LOG_DIR=$(dirname "$LOG_FILE")
if [ ! -d "$LOG_DIR" ]; then
    mkdir -p "$LOG_DIR"
    chmod 755 "$LOG_DIR"
fi

# 日志记录函数
log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" >> "$LOG_FILE"
}

# 调用单个接口函数
call_api() {
    local method=$1
    local url=$2
    local body=$3
    local attempt=1
    local curl_cmd

    # 构造curl命令
    if [ "$method" = "POST" ]; then
        curl_cmd="curl -s -m $TIMEOUT -X POST -H 'Content-Type: application/json' -d '$body' '$url'"
    else
        curl_cmd="curl -s -m $TIMEOUT '$url'"
    fi

    while [ $attempt -le $RETRY_COUNT ]; do
        log "Attempt $attempt: Calling $method API at $url"
        RESPONSE=$(eval "$curl_cmd -w '%{http_code}' -o /tmp/api_response.txt" 2>> "$LOG_FILE")
        HTTP_CODE="${RESPONSE: -3}"  # 提取HTTP状态码

        if [ "$HTTP_CODE" -eq 200 ]; then
            log "Success: HTTP $HTTP_CODE, Response: $(cat /tmp/api_response.txt)"
            return 0
        else
            log "Failed: HTTP $HTTP_CODE, Response: $(cat /tmp/api_response.txt)"
            if [ $attempt -lt $RETRY_COUNT ]; then
                log "Retrying after $RETRY_DELAY seconds..."
                sleep $RETRY_DELAY
            fi
        fi
        ((attempt++))
    done
    log "Error: Failed to call $method API at $url after $RETRY_COUNT attempts"
    return 1
}

# 主逻辑
log "Starting multiple API call script"
EXIT_CODE=0

# 遍历调用所有接口
for api in "${API_CALLS[@]}"; do
    IFS='|' read -r method url body <<< "$api"
    log "Processing $method request to $url"
    call_api "$method" "$url" "$body"
    if [ $? -ne 0 ]; then
        EXIT_CODE=1
    fi
done

# 清理临时文件
rm -f /tmp/api_response.txt

if [ $EXIT_CODE -eq 0 ]; then
    log "Script executed successfully"
else
    log "Script completed with errors"
fi

exit $EXIT_CODE
